cmake_minimum_required (VERSION 3.25)
set(CMAKE_MSVC_DEBUG_INFORMATION_FORMAT "$<IF:$<AND:$<C_COMPILER_ID:MSVC>,$<CXX_COMPILER_ID:MSVC>>,$<$<CONFIG:Debug,RelWithDebInfo>:EditAndContinue>,$<$<CONFIG:Debug,RelWithDebInfo>:ProgramDatabase>>")
project ("easylog" LANGUAGES CXX)
file(GLOB_RECURSE srcs ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp)
set(main_target_name "easylog")
add_library(${main_target_name} SHARED ${srcs})
target_compile_features(${main_target_name} PRIVATE cxx_std_20)
target_compile_definitions(${main_target_name} PRIVATE EASYLOG)
target_include_directories(${main_target_name} PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/../include)
if(CMAKE_CXX_COMPILER_ID MATCHES "GNU|Clang|AppleClang")
    target_compile_options(${main_target_name} PRIVATE -fvisibility=hidden)
endif()
